01 拉取pxc镜像
    docker pull percona/percona-xtradb-cluster

02 复制pxc镜像(实则重命名)
    docker tag percona/percona-xtradb-cluster:latest pxc-cluster

03 删除pxc原来的镜像
    docker rmi percona/percona-xtradb-cluster:latest

04 创建一个单独的网段，给mysql数据库集群使用
     (1)docker network create ace-net

05 创建和删除volume
    创建：docker volume create --name pxc-v1
    创建：docker volume create --name pxc-v2
    创建：docker volume create --name pxc-v3
    删除：docker volume rm v1
    查看详情：docker inspect pxc-v1

最后一歩
docker run -d -p 3311:3306 \
-v pxc-v2:/var/lib/mysql \
-e MYSQL_ROOT_PASSWORD=root \
-e CLUSTER_NAME=PXC \
-e XTRABACKUP_PASSWORD=root \
--privileged --name=pcx-node1 \
--net=ace-net \
pxc-cluster:latest

        [CLUSTER_JOIN将该数据库加入到节点上组成集群]
docker run -d -p 3312:3306 \
-v pxc-v2:/var/lib/mysql \

-e XTRABACKUP_PASSWORD=root \
-e CLUSTER_JOIN=pcx-node1 \
--privileged --name=pcx-node2 \
--net=ace-net \
pxc-cluster:latest

docker run -d -p 3313:3306 \
